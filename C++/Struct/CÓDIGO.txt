////////////////////////////////////////////////////// MÉDIA TOTAL //////////////////////////////////////////////////////

float mediaPT(areas notasPT) {
    int notasModulos = 0;
    int modulos = 0;

    for (int i = 0; i < numModulosPT; i++) {
        if (notasPT.moduloNota[i] >= 10) {
            notasModulos += notasPT.moduloNota[i];
            modulos++;
        }
    }

    if (modulos > 0) {
        return (float)notasModulos / modulos;
    }
    else {
        cout << "No valid module notes were found. Please enter a valid note (10 or above)." << endl;
        return -1;
    }
}
